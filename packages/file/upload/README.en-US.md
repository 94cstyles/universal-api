## Example

```js
import file from 'universal-file';

file.upload({
  url: 'http://httpbin.org/post',
  fileType: 'image',
  fileName: 'file',
  filePath: '**filePath**',
  success: res => {
    console.log('upload success');
  },
  fail: res => {
    console.log('upload succefailss');
  },
});

```
You can also import from the big package:

```js
import { file } from 'universal-api';

```
